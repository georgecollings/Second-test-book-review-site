    </div>


    <!-- Main App -->

    <div id="app-section" class="hidden">

        <h2>Book Review App</h2>


        <div>

            <input type="text" id="book-title" placeholder="Book Title">

            <textarea id="book-review" placeholder="Write your review..."></textarea>

            <input type="number" id="book-rating" min="1" max="10" placeholder="Rating (1-10)">

            <br>

            <input type="file" id="book-cover" accept="image/*" onchange="previewImage(event)">

            <input type="file" accept="image/*" capture="environment" onchange="previewImage(event)">

            <br>

            <img id="preview" class="hidden">

            <br>

            <button onclick="submitReview()">Submit Review</button>

        </div>


        <h3>Reviews</h3>

        <div id="reviews-container"></div>


        <!-- Admin Panel -->

        <h3>Invite Users</h3>

        <input type="email" id="invite-email" placeholder="Enter email to invite">

        <button onclick="inviteUser()">Invite</button>

        <ul id="user-list"></ul>

    </div>


    <script>

        const allowedUsers = JSON.parse(localStorage.getItem('allowedUsers')) || ["user1@example.com", "user2@example.com"];

        const reviews = [];


        function login() {

            const email = document.getElementById("email-input").value;

            if (allowedUsers.includes(email)) {

                document.getElementById("login-section").classList.add("hidden");

                document.getElementById("app-section").classList.remove("hidden");

            } else {

                document.getElementById("login-error").innerText = "Access denied. You are not on the invite list.";

            }

        }


        function inviteUser() {

            const newEmail = document.getElementById("invite-email").value;

            if (newEmail && !allowedUsers.includes(newEmail)) {

                allowedUsers.push(newEmail);

                localStorage.setItem('allowedUsers', JSON.stringify(allowedUsers));

                updateUserList();

                alert(`Invitation sent to ${newEmail}`);

            }

        }


        function updateUserList() {

            const userList = document.getElementById("user-list");

            userList.innerHTML = "";

            allowedUsers.forEach(user => {

                const li = document.createElement("li");

                li.innerText = user;

                userList.appendChild(li);

            });

        }


        function previewImage(event) {

            const reader = new FileReader();

            reader.onload = function () {

                const preview = document.getElementById("preview");

                preview.src = reader.result;

                preview.classList.remove("hidden");

            };

            reader.readAsDataURL(event.target.files[0]);

        }


        function submitReview() {

            const bookTitle = document.getElementById("book-title").value;

            const bookReview = document.getElementById("book-review").value;

            const bookRating = document.getElementById("book-rating").value;

            const bookCover = document.getElementById("preview").src;


            if (bookTitle && bookReview && bookRating) {

                const newReview = { bookTitle, bookReview, bookRating, bookCover };

                reviews.push(newReview);

                displayReviews();

            }

        }


        function displayReviews() {

            const reviewContainer = document.getElementById("reviews-container");

            reviewContainer.innerHTML = "";

            reviews.forEach(review => {

                const reviewCard = document.createElement("div");

                reviewCard.classList.add("review-card");

                reviewCard.innerHTML = `

                    <h4>${review.bookTitle}</h4>

                    <p>${review.bookReview}</p>

                    <p><strong>Rating:</strong> ${review.bookRating}/10</p>

                    ${review.bookCover ? `<img src="${review.bookCover}">` : ""}

                `;

                reviewContainer.appendChild(reviewCard);

            });

        }


        updateUserList();

    </script>
